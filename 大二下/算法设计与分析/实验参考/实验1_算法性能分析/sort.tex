	\begin{algorithm}
		\caption{InsertSort} %算法标题
		\begin{algorithmic}[1] %算法开始
			\FOR{i $\gets$ 1 to length(number)-1}
			\STATE{key$\gets$number[i]}
			\STATE{j$\gets$ i-1}
			\WHILE{$j>= 0$ and $key<number[i]$}
			\STATE{number[j+1]$\gets$number[j - -]}
			\ENDWHILE
			\STATE{number[j+1]$\gets$key}
			\ENDFOR
		\end{algorithmic}
	\end{algorithm}
	
	\begin{algorithm}
		\caption{BubbleSort} %算法标题
		\begin{algorithmic}[1] %算法开始
			\FOR{i $\gets$ 0 to length(number)-2}
			\FOR{j$\gets$length(number)-i-2}
			\IF{number[j]$ > $number[j+1]}
			\STATE{swap number[j] and number[j+1]}
			\ENDIF
			\ENDFOR
			\ENDFOR
		\end{algorithmic}
	\end{algorithm}
	
	\begin{algorithm}
		\caption{SelectSort} %算法标题
		\begin{algorithmic}[1] %算法开始
			\FOR{i $\gets$ 0 to length(number)-2}
			\STATE{k$\gets$i}
			\FOR{j$\gets$i+1 to length(number)-1}
			\IF{number[j]$ < $number[k]}
			\STATE{k$\gets$j}
			\ENDIF
			\ENDFOR			
			\IF{k != i}
			\STATE{swap number[i] and number[k]}
			\ENDIF
			\ENDFOR
		\end{algorithmic}
	\end{algorithm}
	
	\begin{algorithm}
		\caption{MergeSort(low,high)} %算法标题
		\begin{algorithmic}[1] %算法开始
			\IF{low$ < $high}
			\STATE{mid$\gets$(low+high)/2}
			\STATE{MergeSort(low,mid)}
			\STATE{MergeSort(mid+1,high)}
			\STATE{Merge(low,high)}
			\ENDIF
		\end{algorithmic}
	\end{algorithm}
	
	\begin{algorithm}
		\caption{QuickSort(low,high)} %算法标题
		\begin{algorithmic}[1] %算法开始
			\IF{low$ < $high}
			\STATE{pivot$\gets$Divide(low,high)}
			\STATE{QuickSort(low,pivot-1)}
			\STATE{QuickSort(pivot+1,high)}
			\ENDIF
		\end{algorithmic}
	\end{algorithm}
